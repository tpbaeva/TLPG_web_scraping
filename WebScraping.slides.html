Web Scraping

a very practical introduction

Why web scraping

  • it's free
  • data only available on web site
  • scraping faster than other APIs
  • delayed pubication in other APIs

Is it reliable

  • surprisingly so, if done well

How can we scrape in python

  • depends on the web page and/or what you want to achieve
  • inspect source/elements, use eg FireBug
  • old school - urllib2, mechanize, requests + lxml, BeautifulSoup
  • new school - it depends
  • use cookies
  • wrap javascript
  • webdriver - here we will see selenium

def convert_weatherdf(df_data, lon_type='celestial'):
    """ Converts df_data between celestial and signed format by automatically detecting the current format
    """
    df = copy(df_data)
    lons = np.array(df.columns, dtype=float)
    lats = np.array(df.index, dtype=float)   
    if ((min(lons) >= 0) and (lon_type == 'celestial')) or ((min(lons) < 0) and (lon_type == 'signed')):
        return df
    if (min(lons) >= 0) and (lon_type == 'signed'):
        # convert to signed
        lons[lons>180] -= 360
        df = pd.DataFrame(df.values, index=lats, columns=lons)
    if (min(lons) < 0) and (lon_type == 'celestial'):
        # convert to celestial
        lons[lons<0] += 360
    df = pd.DataFrame(df.values, index=lats, columns=lons)
    return df.reindex_axis(sorted(df.columns), axis=1)

import numpy as np
from mpl_toolkits.basemap import Basemap
import matplotlib.pyplot as plt
from copy import copy


def plot_weatherdata(df_data, view, figsize=(10,8), values_adjustemt=-272.15, coastlines=True, countries=True, dispmin=277.5, dispmax=312.5, title=None, clevs=16):
    """ Plots weather data df_data in one of 3 possible views: earth, europe and northhem
    """
    
    if view.upper() == 'EUROPE':
        m = Basemap(projection='cyl', llcrnrlat=30, urcrnrlat=80,
                    llcrnrlon=-20, urcrnrlon=45, resolution='c') 
    elif view.upper() == 'EARTH':
        #m = Basemap(projection='robin',lon_0=0,resolution='c')
        m = Basemap(projection='moll',lon_0=0,resolution='c')
    elif view.upper() == 'NORTHHEM':
        m = Basemap(projection='npstere',boundinglat=5, lon_0=-90,resolution='l')
    else:
        msg = 'Failed to recognize `{}` as a view. Must enter one of `earth`, `northhem` or `europe`.\n'
        raise Exception(msg.format(view))

    df_toplot = copy(df_data)
    df_toplot = df_toplot.reindex_axis(sorted(df_toplot.columns), axis=1)
    lons = np.array(df_toplot.columns, dtype=float)
    lats = np.array(df_toplot.index, dtype=float)
    lons, lats = np.meshgrid(lons, lats)
    x, y = m(lons, lats)
    data = df_toplot.values+values_adjustemt

    fig = plt.figure(figsize=figsize)
    cs = m.pcolor(x, y, data)
    cs = m.contourf(x, y, data, clevs)
    if coastlines:
        m.drawcoastlines()
    if countries:
        m.drawcountries()
    plt.clim(dispmin+values_adjustemt, dispmax+values_adjustemt)
    if view.upper() == 'EUROPE':
        cb = plt.colorbar(shrink=0.75) 
    elif view.upper() == 'EARTH':
        cb = plt.colorbar(shrink=0.5) 
    else:
        cb = plt.colorbar()
    cb.set_label('Surface Air T (C)')
    if title is not None:
        plt.text(0, 0, title, backgroundcolor='white', color='black')
    return fig

def pdate(dateobj, dayfirst=True): 
    """Creates a Python datetime.datetime object from a date-like input 
  
    Parameters 
    ---------- 
    dateobj : date-like object (can be a string, a pandas.Timestamp or pandas.Period)
    dayfirst: if True (default value) converts to dayfirst format
  
    Examples 
    -------- 
    >>> from btglib.utils import pdate
    >>> pdate('25-Nov-2013') 
    datetime.datetime(2013, 11, 25, 0, 0)
    >>> pdate('25/11/2013') 
    datetime.datetime(2013, 25, 11, 0, 0) 
    >>> pdate(pandas.Period(freq='M', year=2013, month=11))
    datetime.datetime(2013, 11, 1, 0, 0)
    """
    
    d = None
    if isinstance(dateobj, datetime): 
        d = dateobj 
    elif isinstance(dateobj, pd.Period): 
        d = dateobj.asfreq('S','S').to_timestamp().to_pydatetime() 
    elif isinstance(dateobj, pd.Timestamp): 
        d = dateobj.to_pydatetime() 
    elif isinstance(dateobj, basestring): 
        if len(dateobj) >= 6: 
            from dateutil.parser import parser
            parser = parser() 
            d = parser.parse(dateobj, dayfirst=dayfirst) 
    else:
        try: 
            year = dateobj.year 
            month = dateobj.month 
            day = dateobj.day 
            hour = getattr(dateobj, 'hour', 0) 
            minute = getattr(dateobj, 'minute', 0) 
            second = getattr(dateobj, 'second', 0) 
            microsecond = getattr(dateobj, 'microsecond', 0) 
            tzinfo = getattr(dateobj, 'tzinfo', None) 
            d = datetime(year, month, day, hour, minute, second, microsecond, tzinfo) 
        except AttributeError: 
            try: 
                year = dateobj.Year 
                month = dateobj.Month 
                day = dateobj.Day 
                hour = getattr(dateobj, 'Hour', 0) 
                minute = getattr(dateobj, 'Minute', 0) 
                second = getattr(dateobj, 'Second', 0) 
                if hasattr(dateobj, 'Millisecond'): 
                    microsecond = 1000*getattr(dateobj, 'Millisecond', 0) 
                else: 
                    microsecond = getattr(dateobj, 'Microsecond', 0) 
                d = datetime(year, month, day, hour, minute, second, microsecond) 
            except AttributeError: 
                pass
    if d is None: 
        msg = 'Could not convert object {} of type {} to a valid date'
        raise ValueError(msg.format(repr(dateobj), type(dateobj))) 
    return d 

def parse_icesource(html):
    """ Parses html of ICE vol page to extact data, using BeautifulSoup
    
    """
    
    chtml = html.replace('<nobr>','')
    chtml = chtml.replace('</nobr>','')
    soup = BeautifulSoup(chtml)
    ssoup = soup('table', {'class': 'table table-data'})

    calls = parse_icetablesoup(ssoup[0])
    puts  = parse_icetablesoup(ssoup[1])
    
    return pd.merge(calls, puts, how='outer')
    
    
def parse_icetablesoup(soup):
    """ Parses beautifulsoup from ICE source to get data
    
    """
    
    columns = [c.string.upper() for c in soup('tr')[0]('th')]
    frame = {}
    tenorkey = None
    for line in soup('tr')[1:]:
        tds = line('td')
        if len(tds) == 2:
            tenorkey = tds[0].string
            frame[tenorkey] = []
        if len(tds) == len(columns):
            values = [c.string for c in line('td')]
            for idx, value in enumerate(values):
                try:
                    values[idx] = float(value)
                except Exception:
                    values[idx] = 0.0
            frame[tenorkey].append(values)
    
    frames = []        
    for key in frame.keys():
        df = pd.DataFrame(frame[key], columns=columns)
        tenor = pdate('1-'+key[:3]+'-'+key[-2:])
        dct = {'Underlying': [tenor]*len(df),
               'Strike' : map(float, df['STRIKE'].values),
               'Vol' : map(float, df['IMPVOL'].values),
               }
        frames.append(pd.DataFrame(dct))    
    return pd.concat(frames, axis=0)

def calcT(dateobj, datelist, factor=365.25):
    """Calculates tenors betwen dateobj and a list of date-like objects
    
       Parameters
       ----------
       dateobj:    string or datetime-like object
                   reference date for tenor calculation
       
       datelist:   list or array of string or datetime-like objects
                   dates to which the tenor is calculated relative to dateobj
       
       factor:     float
                   normalization factor, e.g. 1 for number of days, 365.25 for fractions of
                   calendar year, 252 for fractions of buisiness year etc
       
       Examples
       --------
       >>> from btglib.utils import calcT
       >>> import pandas as pd
       >>> val_date = '6-Feb-2014'
       >>> dates = pd.date_range('10-Feb-2014', '15-Feb-2014', freq='D')
       >>> calcT(val_date, dates)
       array([ 0.0109514 ,  0.01368925,  0.0164271 ,  0.01916496,  0.02190281, 0.02464066])   
    """
    
    d = pdate(dateobj)
    darray = np.atleast_1d(datelist)
    darray = map(pdate, darray)

    res = np.array([(d1-d).total_seconds() for d1 in darray])
    res /= (60.0*60.0*24.0)
    return res/factor
